Порядок внутри процедуры
1. Объявления (IS/AS)
        ↓
2. BEGIN
        ↓
3. Присваивание начальных значений
        ↓
4. SELECT INTO / DML операции
        ↓
5. Условные операторы (IF/CASE)
        ↓
6. Циклы (LOOP/WHILE/FOR)
        ↓
7. Вложенные блоки (если есть)
        ↓
8. Курсорные операции
        ↓
9. EXIT WHEN условия
        ↓
10. COMMIT/ROLLBACK (если есть)
        ↓
11. EXCEPTION (если ошибка)
        ↓
12. END

Элемент	                Правило
IS/AS	                Можно использовать любое
Точка с запятой (;)	    После каждого оператора
Присваивание	        := (двоеточие с равно)
Сравнение	            = (одно равно)
Конкатенация	        || (две вертикальные черты)
Комментарии	            -- однострочный, /* */ многострочный
Кавычки	                Для строк 'текст', для идентификаторов "Имя"
NULL	                Можно присвоить, можно сравнить через IS NULL

CREATE OR REPLACE PROCEDURE comprehensive_example(
    p_in_param NUMBER,
    p_out_param OUT VARCHAR2,
    p_inout_param IN OUT DATE DEFAULT SYSDATE
)
IS
    -- Переменные
    v_local_var VARCHAR2(100);
    v_count NUMBER := 0;
    
    -- Константы
    c_max CONSTANT NUMBER := 1000;
    
    -- Курсоры
    CURSOR c_data IS SELECT id FROM dishes;
    
    -- Типы
    TYPE t_rec IS RECORD (id NUMBER, name VARCHAR2(100));
    v_rec t_rec;
    
    -- Пользовательское исключение
    e_bad_value EXCEPTION;
    
    -- Вложенная функция
    FUNCTION validate_price(p_price NUMBER) RETURN BOOLEAN IS
    BEGIN
        RETURN p_price > 0 AND p_price < c_max;
    END;
    
BEGIN
    -- Основной код
    SAVEPOINT sp_start;
    
    -- Присваивание
    v_local_var := 'START';
    
    -- SELECT INTO
    SELECT COUNT(*) INTO v_count FROM dishes;
    
    -- Условия
    IF v_count = 0 THEN
        RAISE e_bad_value;
    END IF;
    
    -- Цикл
    FOR i IN 1..10 LOOP
        EXIT WHEN i > v_count;
        
        BEGIN
            -- Вложенный блок с обработкой ошибок
            NULL;
        EXCEPTION
            WHEN OTHERS THEN
                NULL;
        END;
    END LOOP;
    
    -- DML
    UPDATE dishes SET price = price * 1.1;
    v_count := SQL%ROWCOUNT;
    
    -- OUT параметр
    p_out_param := 'Успешно: ' || v_count;
    
    COMMIT;
    
EXCEPTION
    WHEN e_bad_value THEN
        ROLLBACK TO sp_start;
        p_out_param := 'Ошибка валидации';
        
    WHEN OTHERS THEN
        ROLLBACK;
        p_out_param := 'Ошибка: ' || SQLERRM;
        RAISE;
END comprehensive_example;
